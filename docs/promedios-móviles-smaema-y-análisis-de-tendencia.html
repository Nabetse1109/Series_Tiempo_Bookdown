# --- 1. Calcular SMA y EMA ---
EMA_classic <- function(x, n){
  alpha <- 2/(n + 1)
  y <- rep(NA_real_, length(x))
  idx <- which(!is.na(x))
  if(length(idx) == 0) return(y)
  y[idx[1]] <- x[idx[1]]
  for(i in idx[-1]) y[i] <- alpha * x[i] + (1 - alpha) * y[i - 1]
  y
}

k_short <- 6    # EMA(6)
